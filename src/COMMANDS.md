# Supported actions

## Batch 1 (implemented)
- create_track(type)
- select_track(target, value)
- load_device(track, device)
- arm_track(track, value)
- set_monitor(track, mode)
- transport_play()
- transport_stop()
- transport_record(value)
- session_record(value)
- wait(ms)

## Batch 2 (implemented)
- create_scene(index)
- fire_scene(scene)
- stop_scene(scene)
- fire_clip(track, slot)
- stop_clip(track, slot)
- duplicate_clip(track, slot)
- delete_clip(track, slot)
- set_clip_loop(track, slot, start, end, enabled)

## Batch 3 (implemented)
- set_volume(track, value)
- set_pan(track, value)
- set_send(track, send, value)
- mute_track(track, value)
- solo_track(track, value)

## Batch 4 (implemented)
- list_devices(track)
- set_device_parameter(track, device, parameter, value)
- enable_device(track, device, value)

## Batch 5 (implemented)
- browser_search(query)
- browser_load_to_track(item, track)
- browser_load_preset(device, preset)

## Batch 6 (implemented)
- set_arrangement_position(value)
- set_loop_region(start, end, enabled)
- set_punch_in(value)
- set_punch_out(value)

## Batch 7 (implemented)
- set_tempo(value)
- set_time_signature(numerator, denominator)
- set_metronome(value)
- set_global_quantization(value)

## Batch 8 (implemented)
- project_new(confirm)
- project_save(confirm)
- project_save_as(path, confirm)

## Batch 9 (implemented)
- watch_song(fields)
- watch_track(track, fields)
- watch_list()
- unwatch(id)

## Batch 10 (implemented)
- run_macro(name)
- set_confirmation_policy(value)
